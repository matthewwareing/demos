## Usage

```tsx
type TabData = {  id: string;  title: string;  // other data properties};
// Basic usageconst data: TabData[] = [  { id: '1', title: 'Tab 1' },  { id: '2', title: 'Tab 2' },  // ...more tabs];
<DynamicTabIndicator data={data} />;
```
